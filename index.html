        // Function to construct and display the prompt in a textarea

        function generatePrompt(methodId) {

            let promptText = "";

            const section = document.getElementById(methodId);


            // Select the main action input using a more specific ID pattern

            const actionInput = section.querySelector(`input[id="${methodId}-action-main"]`);

            

            // Select all component inputs (text and textarea) excluding the main action input and the generated prompt textarea

            const componentInputs = section.querySelectorAll(

                `.prompt-component input[type="text"]:not([id="${methodId}-action-main"]), .prompt-component textarea:not([id="${methodId}-generated-prompt"])`

            );

            

            // Select the output textarea using a specific ID pattern

            const outputTextarea = section.querySelector(`textarea[id="${methodId}-generated-prompt"]`);


            if (!actionInput || actionInput.value.trim() === "") {

                alert("Vui lòng nhập 'Action (Hành động chính)' để tạo prompt.");

                if(outputTextarea) outputTextarea.value = ""; // Clear previous prompt if any

                return;

            }


            promptText = actionInput.value.trim(); // Start with the main action


            let componentsContent = "";

            componentInputs.forEach(input => {

                // Ensure we are not processing the output textarea itself if it somehow got selected (e.g. if it had .prompt-component class directly)

                if (input.id === `${methodId}-generated-prompt`) return;

                

                // Also, for I-D-E-A, S-T-E-P, A-C-E, G-O-A-L, the component labels are slightly different

                // We need to ensure the label is correctly fetched.

                const labelElement = section.querySelector(`label[for="${input.id}"]`);

                

                if (labelElement && input.value.trim() !== "") {

                    // Get the text before the colon, or the full text if no colon.

                    const labelText = labelElement.textContent.split(':')[0].trim();

                    componentsContent += `\n${labelText}: ${input.value.trim()}`;

                }

            });


            if (componentsContent) {

                promptText += "\n" + componentsContent; // Add a newline only if there are components

            }


            if(outputTextarea) {

                outputTextarea.value = promptText.trim();

            } else {

                console.error("Output textarea not found for methodId: " + methodId);

            }

        }

    </script>
